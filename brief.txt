let selectedSubcategory = null;
let currentFee = 0;

console.log('Script loaded. Initial state:', { selectedCategory, selectedSubcategory });

// Platform fee tiers
const platformFees = {
  // $25 tier
  'YouTube': 25,
  'TikTok': 25,
  'Instagram': 25,
  'Roku': 25,
  'Hulu': 25,
  'Pluto TV': 25,
  'Samsung TV+': 25,
  'Twitch': 25,
  'Fortnite': 25,
  'Roblox': 25,
  // $15 tier
  'Facebook': 15,
  'LinkedIn': 15,
  'Twitter/X': 15,
  'Snapchat': 15,
  'Discord': 15,
  'Steam': 15,
  // Default $10 for all others
};

function getPlatformFee(platform) {
  return platformFees[platform] || 10;
}

function updateFeeDisplay(category, platform) {
  const feeDisplay = document.getElementById(`fee-${category}`);
  if (feeDisplay && platform) {
    const fee = getPlatformFee(platform);
    feeDisplay.querySelector('.fee-value').textContent = fee;
    feeDisplay.classList.add('visible');
    currentFee = fee;
  } else if (feeDisplay) {
    feeDisplay.classList.remove('visible');
  }
}

function selectCard(card) {
console.log('selectCard clicked:', card.dataset.category);

const category = card.dataset.category;
const allCards = document.querySelectorAll('.gem-card');

// If clicking already selected card with subcategory, deselect everything
if (selectedCategory === category && selectedSubcategory) {
  console.log('Deselecting everything');
  selectedCategory = null;
  selectedSubcategory = null;
  currentFee = 0;
  allCards.forEach(c => {
    c.classList.remove('selected', 'locked');
    // Hide fee displays
    const feeDisp = c.querySelector('.fee-display-card');
    if (feeDisp) feeDisp.classList.remove('visible');
    // Reset dropdowns
    const select = c.querySelector('.category-select');
    if (select) select.value = "";
  });
  document.getElementById('continueContainer').classList.remove('visible');
  return;
}

// If clicking a new card while another is selected, switch selection
if (selectedCategory && selectedCategory !== category) {
  // Clear previous selections
  allCards.forEach(c => {
    c.classList.remove('selected', 'locked');
    const feeDisp = c.querySelector('.fee-display-card');
    if (feeDisp) feeDisp.classList.remove('visible');
    const select = c.querySelector('.category-select');
    if (select && c !== card) select.value = "";
  });
  selectedSubcategory = null;
  currentFee = 0;
}

// Select new category
selectedCategory = category;

// LOCK ALL OTHER CARDS immediately
allCards.forEach(c => {
  if (c === card) {
    c.classList.add('selected');
    c.classList.remove('locked');
    // Focus the dropdown
    const select = c.querySelector('.category-select');
    if (select) {
      setTimeout(() => select.focus(), 100);
    }
  } else {
    c.classList.add('locked');
    c.classList.remove('selected');
  }
});

// Clear any previous subcategory selection when clicking new card
if (!selectedSubcategory || selectedCategory !== category) {
  selectedSubcategory = null;
  updateFeeDisplay(category, null);
}

checkContinue();
}

function updateSelectionFromEvent(selectElement, category) {
const value = selectElement.value;
console.log('Dropdown changed:', category, value);

if (value && value !== "") {
  selectedCategory = category;
  selectedSubcategory = value;
  
  // Calculate and display fee
  updateFeeDisplay(category, value);
  
  // LOCK ALL OTHER CARDS IMMEDIATELY
  const allCards = document.querySelectorAll('.gem-card');
  allCards.forEach(card => {
    const cardCategory = card.dataset.category;
    if (cardCategory === category) {
      card.classList.add('selected');
      card.classList.remove('locked');
    } else {
      card.classList.add('locked');
      card.classList.remove('selected');
      // Hide fee displays on other cards
      const feeDisp = card.querySelector('.fee-display-card');
      if (feeDisp) feeDisp.classList.remove('visible');
    }
  });
  
  console.log('Selection confirmed:', { selectedCategory, selectedSubcategory, currentFee });
  checkContinue();
} else {
  console.log('Empty selection, clearing subcategory');
  selectedSubcategory = null;
  currentFee = 0;
  updateFeeDisplay(category, null);
  checkContinue();
}
}

function checkContinue() {
console.log('Checking continue button...', { selectedCategory, selectedSubcategory });

const hasCategory = selectedCategory !== null && selectedCategory !== "";
const hasSubcategory = selectedSubcategory !== null && selectedSubcategory !== "";
const continueContainer = document.getElementById('continueContainer');

if (hasCategory && hasSubcategory) {
  console.log('Showing continue button');
  continueContainer.classList.add('visible');
} else {
  console.log('Hiding continue button');
  continueContainer.classList.remove('visible');
}
}

function goToStage2() {
console.log('goToStage2 called', { selectedCategory, selectedSubcategory, currentFee });

if (!selectedCategory || !selectedSubcategory) {
  console.error('Cannot proceed - missing selection', { selectedCategory, selectedSubcategory });
  alert('Please select a category and platform first');
  return;
}

// Apply category styling to big card
const card = document.getElementById('stage2Card');
card.className = 'big-card ' + selectedCategory;

// Transition
document.getElementById('stage1').classList.add('hidden');
document.getElementById('stage2').classList.remove('hidden');

console.log('Transitioned to Stage 2');
}

function backToStage1() {
console.log('Going back to Stage 1');
document.getElementById('stage2').classList.add('hidden');
document.getElementById('stage1').classList.remove('hidden');
}

console.log('goToStage3 called');

const budget = document.getElementById('budgetInput').value;
const geo = document.getElementById('geoSelect').value;

const gender = document.getElementById('genderSelect').value;

// Apply category styling

// Populate summary
const categoryNames = {

document.getElementById('summaryCategory').textContent = categoryNames[selectedCategory];

}

function backToStage2() {

document.getElementById('stage3').classList.add('hidden');
document.getElementById('stage2').classList.remove('hidden');
}

async function saveCampaign() {
console.log('Saving campaign...');

const campaignData = {
  category: selectedCategory,
  subcategory: selectedSubcategory,
  budget: parseInt(document.getElementById('budgetInput').value),
  target_views: parseInt(document.getElementById('targetViews').value),
  geography: document.getElementById('geoSelect').value,
  age_range: document.getElementById('ageSelect').value,
  gender: document.getElementById('genderSelect').value,
  content_criteria: document.getElementById('contentSelect').value,
  brief: document.getElementById('briefText').value,
  auction_fee: currentFee,
  created_at: new Date().toISOString(),
  status: 'pending'
};

console.log('Data to save:', campaignData);

try {
  const { data, error } = await supabaseClient
    .from('campaigns')
    .insert([campaignData]);
  
  if (error) throw error;
  
  console.log('Save successful');
  showToast('Campaign saved successfully');
} catch (err) {
  console.error('Save failed:', err);
  showToast('Error saving campaign');
}
}

function showToast(message) {
const toast = document.getElementById('toast');
toast.textContent = message;
toast.classList.add('visible');
setTimeout(() => {
  toast.classList.remove('visible');
}, 3000);
}

console.log('Initializing...');
